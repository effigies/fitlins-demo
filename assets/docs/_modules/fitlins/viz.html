        fig = plt.figure()
        axes = plt.gca()
        if 'axes' in kwargs:
            kwargs['axes'] = axes
        else:
            kwargs['ax'] = axes
    plotter(*args, **kwargs)
    fig.savefig(fname, bbox_inches='tight')
    plt.close(fig)
